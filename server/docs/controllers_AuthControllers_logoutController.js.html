import { ApiResponse } from "#src/utils/ApiResponse";

/**
 * <h3>Function to log out the user</h3>
 * * <b>POST REQUEST</b>
 * <p>This function verify user's credentials and delete jwt cookie to log out the user from app</p>
 *
 * @async
 * @param {Request} req request sent from client
 * @param {Response} res response generated from server
 * @param {NextFunction} next next functionality
 * @returns {Response}
 * <p><i>any server error -> 500 ("INTERNAL SERVER ERROR")</i></p>
 * <p><i>jwt cookie deleted -> 200 ("LOGOUT SUCCESSFULLY")</i></p>
 */
export const logOut = async (req, res) => {
	// creating instance of ApiResponse class
	const customResponse = new ApiResponse(res);

	try {
		// Setting up the token cookie in client's machine
		// res.cookie("jwt", "", {
		// 	maxAge: 1,
		// 	secure: true,
		// 	sameSite: "None",
		// 	httOnly: true,
		// });

		res.clearCookie("jwt");

		// Successful response
		return customResponse.send({ statusCode: 200, message: "LOGOUT SUCCESSFULLY" });
	} catch (error) {
		// Logging the error details
		console.log("ðŸš€ ~ Logout ~ error:", error);
		return customResponse.send({ statusCode: 500, message: "INTERNAL SERVER ERROR" });
	}
};
