import jwt from "jsonwebtoken";
import { JWT_KEY, JWT_TOKEN_AGE } from "./constants.js";

/**
 * <h3>This function takes user's email and id and generate jwt access token for specified time period</h3>
 *
 * @param {string} email user's email id
 * @param {string} userId user's mongodb database id
 * @returns {string} jwt token with user's details
 */
export const jwtTokenGenerator = (email, userId) =>
	jwt.sign({ email, userId }, JWT_KEY, { expiresIn: JWT_TOKEN_AGE });
