import User from "#src/models/UserModel";
import logger from "#src/utils/logger";
import { ErrorResponse } from "#src/utils/responses/ErrorResponse";
import { SuccessResponse } from "#src/utils/responses/SuccessResponse";

/**
 * <h3>API to provide user's details</h3>
 *
 *
 * @async
 * @function getUserInfo
 * @param {Object} req - Express request object.
 * @param {Object} res - Express response object.
 *
 * @returns {SuccessResponse<Object>} - Sends a response containing user data.
 * @returns {ErrorResponse<Object>} - Sends a response containing an error details.
 *
 * @description
 * This function fetches user information from the database based on the user ID set by middleware.
 * <br/><b>`GET REQUEST`</b><br/>
 * Status codes returns:
 * - 404 for user not found.
 * - 200 for successful response.
 * - 500 for server error or any other error.
 */

export const getUserInfo = async (req, res) => {
	try {
		// Fetching the user data from database using id set by middleware
		const user = await User.findById(req.userId)
			.select(["-password", "-updatedAt", "-createdAt"])
			.lean();

		if (!user) {
			return new ErrorResponse(res).send({ statusCode: 404, message: "User not fond" });
		}

		// Successful response to return the user's data
		return new SuccessResponse(res).send({ data: user });
	} catch (error) {
		logger.error(error);
		return new ErrorResponse(res).send({ error: [error], message: error.message });
	}
};
