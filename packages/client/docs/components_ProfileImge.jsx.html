import { Avatar, AvatarImage, AvatarFallback } from "@/components/ui/avatar";
import { useRef } from "react";
/**
 * <h3>Profile Image component</h3>
 * <b>This component is responsible to display profile images.</b><br/>
 * <p>This component uses input field to accept file input and save it as base64 string</p>
 * <p>This component uses shadcn Avatar to display uploaded image</p>
 *
 *
 * @param {string} props user's profile image as base64 string
 * @returns Avatar of profile image
 */
const ProfileImage = (props) => {
	const inputRef = useRef();
	const { image, setImage } = { ...props };
	const handleClick = () => inputRef.current.click();

	/**
	 * <h3>Function for file upload</h3>
	 * <p>This function handle image uploaded by user then converts it in base64 string and displays to user</p>
	 *
	 * @param {*} event event occurs on input files
	 */
	const handleChange = (event) => {
		const file = event.target.files[0];
		if (file) {
			const reader = new FileReader();
			reader.onloadend = () => {
				const encodedData = reader.result;
				setImage(encodedData);
			};
			reader.readAsDataURL(file);
		}
	};

	return (
		<div className="w-[200px] h-[200px] rounded-full cursor-pointer">
			{/* ----------------------------------------------------------
				* Image input 
				---------------------------------------------------------- */}
			<input
				type="file"
				className="hidden"
				ref={inputRef}
				onChange={handleChange}
				accept=".png, .jpg,. jpeg"
			/>

			{/* ----------------------------------------------------------
				* Avatar
				---------------------------------------------------------- */}
			<Avatar className="w-full h-full" onClick={handleClick}>
				<AvatarImage src={image ? image : ""} alt="image" />
				<AvatarFallback>USER</AvatarFallback>
			</Avatar>
		</div>
	);
};

export default ProfileImage;
